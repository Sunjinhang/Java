public static void SelectionSort(int [] array) {
		for(int i = 0; i < array.length; i++) {
			int index = i;
			for(int j = i + 1;j < array.length ; j++) {
				if(array[j] < array[index]) {
					index = j;
				}
			}
			if(i == index)
			{
				continue;
			}
			else {
				int temp = array[i];
				array[i] = array[index];
				array[index] = temp;
			}
		}
