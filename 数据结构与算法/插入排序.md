public static void InsertSort(int [] array) {
		for(int i = 1; i < array.length ;i++) {
			int index = i - 1;
			int temp = array[i];
			while(index >=0 && array[index] > temp )
			{
				array[index + 1] = array[index];
				index--;
			}
			array[index + 1] = temp;
		}
